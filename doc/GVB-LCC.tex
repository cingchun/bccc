    \setlength{\parskip}{0.5\baselineskip}
    GVB-BCCC wave function:$\Psi^{GVB-BCCC}=e^{\hat{T}}\vert\Phi _{0}^{}\rangle$,
    $\Phi_{0}^{}$ is GVB wave function
    \newline
    $
        \hat{H}\Psi^{GVB-BCCC}=E\Psi^{GVB-BCCC}\Rightarrow
        \hat{H}e^{\hat{T}}\vert\Phi _{0}^{}\rangle=Ee^{\hat{T}}\vert\Phi _{0}^{}\rangle\Rightarrow
        e^{\hat{-T}}\hat{H}e^{\hat{T}}\vert\Phi _{0}^{}\rangle=E\vert\Phi _{0}^{}\rangle
    $
    \newline
    $\left\{
        \begin{aligned}
            &\langle\Phi_{0}^{}\vert{e}^{\hat{-T}}\hat{H}e^{\hat{T}}\vert\Phi _{0}^{}\rangle=E \qquad\text{Energy equation}\\
            &\langle\Phi_{a}^{}\vert{e}^{\hat{-T}}\hat{H}e^{\hat{T}}\vert\Phi _{0}^{}\rangle=0 \qquad\text{Amplitude Equation}
        \end{aligned}
    \right.$
    \newline
    where $\vert\Phi_{a}^{}\rangle$ is $a$th excited state, and excited states
    \newline
    $\psi_{A}^{}=\hat{T}\Phi _{0}^{}=\displaystyle\sum_{a=1}^{n_{A}}t_{a}\vert\Phi_{a}^{}\rangle$
    \newline
    \newline
    linear CC expansion: $e^{\hat{-T}}\hat{H}e^{\hat{T}}=\hat{H}+[\hat{H},\hat{T}]\qquad [\hat{H},\hat{T}]=\hat{H}\hat{T}-\hat{T}\hat{H}$
    \newline
    $
        E=\langle\Phi_{0}^{}\vert{e}^{\hat{-T}}\hat{H}e^{\hat{T}}\vert\Phi _{0}^{}\rangle =\langle\Phi _{0}^{}\vert\hat{H}+\left[\hat{H},\hat{T}\right]\vert\Phi _{0}^{}\rangle \\
        =\langle\Phi _{0}^{}\vert\hat{H}\vert\Phi _{0}^{}\rangle+\langle\Phi _{0}^{}\vert\left[\hat{H},\hat{T}\right]\vert\Phi_{0}^{}\rangle
        =\langle\Phi _{0}^{}\vert\hat{H}\vert\Phi _{0}^{}\rangle+\langle\Phi _{0}^{}\vert\hat{H}\hat{T}\vert\Phi _{0}^{}\rangle-\langle\Phi _{0}^{}\vert\hat{T}\hat{H}\vert\Phi _{0}^{}\rangle \\
        =\langle\Phi _{0}^{}\vert\hat{H}\vert\Phi _{0}^{}\rangle+\langle\Phi _{0}^{}\vert\hat{H}\hat{T}\vert\Phi _{0}^{}\rangle
        =\langle\Phi _{0}^{}\vert\hat{H}\vert\Phi _{0}^{}\rangle+\langle\Phi _{0}^{}\vert\hat{H}\vert\psi _{A}^{}\rangle \\
        =E_{0}+\displaystyle\sum_{a=1}^{n_{A}}t_{a}\langle\Phi _{0}^{}\vert \hat{H}\vert\Phi _{a}^{}\rangle
    $
    \newline
    Correlation Energy:
    $\Delta{E}=E-E_{0}=\displaystyle\sum_{a=1}^{n_{A}}t_{a}\langle\Phi _{0}^{}\vert \hat{H}\vert\Phi _{a}^{}\rangle$
    \newline
    \newline
    $
        0=\langle\Phi _{a}^{}\vert e^{\hat{-T}}\hat{H}e^{\hat{T}}\vert\Phi _{0}^{}\rangle
        =\langle\Phi _{a}^{}\vert \left[\hat{H},\hat{T}\right]\vert\Phi _{0}^{}\rangle \\
        =\langle\Phi _{a}^{}\vert \hat{H}\vert\Phi _{0}^{}\rangle+\langle\Phi _{a}^{}\vert \hat{H}\hat{T}\vert\Phi _{0}^{}\rangle-\langle\Phi _{a}^{}\vert \hat{T}\hat{H}\vert\Phi _{0}^{}\rangle \\
        =\langle\Phi _{a}^{}\vert \hat{H}\vert\Phi _{0}^{}\rangle+\langle\Phi _{a}^{}\vert \hat{H}\vert\psi _{A}^{}\rangle-\langle\Phi _{a}^{}\vert \hat{T}\hat{H}\vert\Phi _{0}^{}\rangle \\
        =\langle\Phi _{a}^{}\vert \hat{H}\vert\Phi _{0}^{}\rangle+\displaystyle\sum_{b=1}^{n_{A}}t_{b}\langle\Phi _{a}^{}\vert \hat{H}\vert\Phi_{b}^{}\rangle
            -\langle\Phi _{a}^{}\vert \hat{T}\hat{H}\vert\Phi _{0}^{}\rangle
    $
    \newline
    \newline
    $
        \langle\Phi _{a}^{}\vert \hat{T}\hat{H}\vert\Phi _{0}^{}\rangle
            =\langle\Phi _{a}^{}\vert \hat{T}
                \left(\displaystyle\sum_{b\neq 0}^{n_{A}}\vert\Phi_{b}^{} \rangle\langle\Phi_{b}^{}\vert+\vert\Phi _{0}^{} \rangle\langle\Phi _{0}^{}\vert\right)
                \hat{H}\vert\Phi _{0}^{}\rangle \\
            =\langle\Phi _{a}^{}\vert \hat{T}\displaystyle\sum_{b\neq 0}^{n_{A}}\vert\Phi_{b}^{} \rangle\langle\Phi_{b}^{}\vert\hat{H}\vert\Phi _{0}^{}\rangle
                +\langle\Phi _{a}^{}\vert \hat{T}\vert\Phi _{0}^{} \rangle\langle\Phi _{0}^{}\vert\hat{H}\vert\Phi _{0}^{}\rangle \\
            =\displaystyle\sum_{b\neq 0}^{n_{A}}\langle\Phi _{a}^{}\vert \hat{T}\vert\Phi_{b}^{} \rangle\langle\Phi_{b}^{}\vert\hat{H}\vert\Phi _{0}^{}\rangle
                +E_{0}^{}\langle\Phi _{a}^{}\vert\hat{T}\vert\Phi _{0}^{}\rangle
    $
